#include <stdio.h>

int main() {
 int n=5, i, j, s = 1;   
  
    
    for (i = 1; i <= 5; i++)  
    {  
        for (j = 1; j <= i; j++)  
        {  
            printf( "* ");  
        }  
        printf("\n");  
    }  
    for (i = n-1; i >= 1; i--)  
    {  
        for (j = 1; j <= i; j++)  
        {  
            printf ("* ");  
        }  
        printf("\n");  
    }  
   return 0;
}
